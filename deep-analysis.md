```cpp
1. 栈分配优先: 小对象使用栈内存
2. 预分配策略: 缓冲区预分配避免重复分配
3. 零拷贝设计: 使用string_view避免字符串复制

// 实际示例
auto process_container(const T& container) 
    -> std::vector<typename T::value_type> {
    
    // 预分配结果向量
    std::vector<typename T::value_type> result;
    result.reserve(container.size());  // 关键优化
    
    for (const auto& item : container) {
        result.push_back(std::forward<decltype(item)>(item));
    }
    
    return result;
}
```
